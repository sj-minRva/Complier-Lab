

%{
int vow_count=0;
int const_count=0;
%}
vowel [aeiouAEIOU]
consonant [a-zA-Z]
%%
{vowel} {vow_count++;}
[a-zA-Z] {const_count++;}
%%

int main(){
printf("Enter string of vowels and consonant:\n");
yylex();
printf("No of vowels are: %d\n", vow_count);
printf("No of consonants are: %d\n", const_count);
return 0;
}

int yywrap()
{
return 1;
}
